CREATE VIEW empdeptview AS 
SELECT e.EMPNAME, e.DOB, e.SALARY, d.DEPTID, d.DEPTNAME, d.DEPTLOC
FROM Employee e, Department d 
WHERE e.deptid = d.DEPTID AND e.deptid = 5;
